# 代理模式

- 使用者无权访问目标对象
- 中间加代理，通过代理做授权和控制

## 示例

- 科学上网
- 明星经纪人

![类图](https://github.com/viivLgr/viivBlog/blob/master/images/uml-9.png)

```javascript
class ReadImg {
    constructor(fileName) {
        this.fileName = fileName
        this.loadFromDisk()
    }
    display() {
        console.log('display... ' + this.fileName)
    }
    loadFromDisk() {
        console.log('loading... ' + this.fileName)
    }
}

class ProxyImg {
    constructor(fileName) {
        this.realImg = new ReadImg(fileName)
    }
    display() {
        this.realImg.display()
    }
}

// 测试
let proxyImg = new ProxyImg('1.png')
proxyImg.display()
// loading... 1.png
// display... 1.png
```

```html
        // 原生js 判断target
        var div1 = document.getElementById('div1')
        div1.addEventListener('click', function(e) {
            var target = e.target
            if(target.nodeName === 'A'){
                alert(target.innerHTML)
            }
        })

        // setTimeout 改变this指向
        $('#div1').click(function() {
            setTimeout(function() {
                console.log('this', $(this)) // 指向window
                $(this).css('backgroundColor', 'yellow')
            }, 1000)
        })

        // 使用变量保留this
        $('#div1').click(function() {
            var _this = this
            setTimeout(function() {
                console.log('this', $(_this)) // div1
                $(_this).css('backgroundColor', 'yellow')
            }, 1000)
        })

        // 使用jQuery的$.proxy 设置this
        $('#div1').click(function() {
            var fn = function() {
                console.log('this', $(this))
                $(this).css('backgroundColor', 'yellow')
            }
            fn = $.proxy(fn, this) // 改变this指向
            setTimeout(fn, 1000)

            // setTimeout($.proxy(function(){
            //     $(this).css('backgroundColor', 'yellow')
            // }, this), 1000)
        })
        // 使用jQuery的$.proxy 设置this
        $('#div1').click(function() {
            setTimeout($.proxy(function(){
                $(this).css('backgroundColor', 'yellow')
            }, this), 1000)
        })
    </script>
</body>
</html>
```

- ES6 Proxy

明星经纪人

```javascript
// 明星
let star = {
    name: 'seek',
    age: 25,
    phone: 'star: 13212341234'
}

// 经纪人
let agent = new Proxy(star, {
    get: (target, key) => {
        if(key === 'phone') {
            return 'agent: 13211111111' // 经纪人自己的电话
        }
        if(key === 'price') {
            // 明星不报价，经纪人报价
            return 120000
        }
        return target[key]
    },
    set: (target, key, val) => {
        if(key === 'customPrice') {
            if(val < 10000) {
                throw new Error('价格太低')
            }else{
                target[key] = val
            }
        }
        return true
    }
})

console.log(agent.name)
console.log(agent.age)
console.log(agent.phone)
console.log(agent.price)
agent.customPrice = 19000
console.log('agent', agent)
```
